## Code

```java
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode ans = new ListNode(-1);
        ListNode curr = ans;
        int carry = 0;
        while (l1 != null || l2 != null) {
            int sum = 0;
            sum += carry;
            if (l1 != null) {
                sum += l1.val;
                l1 = l1.next;
            }
            if (l2 != null) {
                sum += l2.val;
                l2 = l2.next;
            }
            int newNodeVal = sum % 10;
            ListNode newNode = new ListNode(newNodeVal);
            carry = sum / 10;
            curr.next = newNode;
            curr = curr.next;
        }

        if (carry == 1) {
            ListNode newNode = new ListNode(carry);
            curr.next = newNode;
            curr = curr.next;
        }

        return ans.next;
    }
}
```

```java
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        Stack<Integer> s1 = new Stack<>();
        Stack<Integer> s2 = new Stack<>();
        while (l1 != null) {
            s1.push(l1.val);
            l1 = l1.next;
        }
        while (l2 != null) {
            s2.push(l2.val);
            l2 = l2.next;
        }
        ListNode ans = new ListNode(0); // 0 is act like carry here

        while (!s1.isEmpty() || !s2.isEmpty()) {
            int sum = 0;
            sum += ans.val;
            if (!s1.isEmpty()) {
                sum += s1.pop();
            }

            if (!s2.isEmpty()) {
                sum += s2.pop();
            }

            ans.val = sum % 10;
            ListNode newNode = new ListNode(sum / 10);
            newNode.next = ans;
            ans = newNode;
        }

        return ans.val == 0 ? ans.next : ans;
    }
}
```

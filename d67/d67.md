## Using (Tabulation) Bottom-Up Approach :

```java
class Solution {
    //meaning: dp[i][j] = Do have i jth sum at ith index ?
    private boolean tabulation(int[] arr, int sum) {
        int n = arr.length;
        boolean[][] dp = new boolean[n + 1][sum + 1];

        // Base cases

        // subset with sum 0 is always possible
        for (int i = 0; i <= n; i++) {
            dp[i][0] = true;
        }
        // i >= n (out of bound condition)
        for (int j = 1; j <= sum; j++) {
            dp[n][j] = false;
        }

        for (int i = n - 1; i >= 0; i--) {
            for (int j = 0; j <= sum; j++) {
                boolean take = false;
                if (j >= arr[i]) {
                    take = dp[i + 1][j - arr[i]];
                }
                boolean skip = dp[i + 1][j];
                dp[i][j] = take || skip;
            }
        }
        return dp[0][sum];
    }

    public boolean canPartition(int[] nums) {
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }

        if (sum % 2 != 0) {
            return false;
        }

        int targetSum = sum / 2;

        return tabulation(nums, targetSum); // subset sum problem
    }
}
/*
Time Complexity: O(n*sum)
Space Complexity: O(n*sum)
*/
```
